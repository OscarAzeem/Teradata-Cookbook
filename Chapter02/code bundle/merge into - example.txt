CREATE TABLE MergingTable_Source (
ID   DECIMAL(18,0)
,END_DT DATE
,SOLD_AMT
)
UNIQUE PRIMARY INDEX (ID);

CREATE TABLE MergingTable_Target ( 
ID   DECIMAL(18,0),
END_DT DATE
,SOLD_AMT
)
UNIQUE PRIMARY INDEX (ID);

MERGE INTO MergingTable_Target t
USING MergingTable_Source       s
ON T.ID = S.ID
WHEN MATCHED THEN
UPDATE SET END_DT = S.END_DT ,
SOLD_AMT= S.SOLD_AMT
WHEN NOT MATCHED THEN
INSERT (S.ID, 
S.END_DT, 
S.SOLD_AMT);
